def operand_decider(num, operand):
    if operand == None:
        return num
    if operand[0] == '+':
        return num + operand[1]
    elif operand[0] == '-':
        return num - operand[1]
    elif operand[0] == '*':
        return num * operand[1]
    elif operand[0] == '/':
        return int(num / operand[1])
def zero(operand = None): return operand_decider(0, operand)
