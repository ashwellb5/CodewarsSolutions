def solution(n):
    roman_dict = {
    1 : "I",
    4 : "IV",
    5 : "V",
    9 : "IX",
    10 : "X",
    40 : "XL",
    50 : "L",
    90 : "XC",
    100 : "C",
    400 : "CD",
    500 : "D",
    900 : "CM",
    1000 : "M"
    }
    roman_solution = ""
    while n > 0:
        temp = max(x for x in roman_dict if x <= n)
        n -= temp
        roman_solution += roman_dict[temp]
    return roman_solution
