def thirt(n):
    n_str = str(n)
    i = 0
    tens_list = []
    while i < len(n_str):
        tens_list.append((10 ** i) % 13)
        i+=1
    i = 0
    sum = 0
    while i < len(tens_list):
        sum += tens_list[i] * int(n_str[-(i+1)])
        i+=1
    if sum == n:
        return sum
    else:
        return thirt(sum)
